45 uint8_t can_init(uint32_t can_br, uint8_t num_tx_mb, uint8_t num_rx_mb)
46 {
47 
48  //Make sure num_rx_mb and num_tx_mb is valid
49  if(num_rx_mb > 8 | num_tx_mb > 8 | num_rx_mb + num_tx_mb > 8)
50  {
51  return 1; //Too many mailboxes is configured
52  }
53 
54 
55  uint32_t ul_status;
56 
57  //Disable can
58  CAN0->CAN_MR &= ~CAN_MR_CANEN;
59  //Clear status register on read
60  ul_status = CAN0->CAN_SR;
61 
62 
63  // Disable interrupts on CANH and CANL pins
64  PIOA->PIO_IDR = PIO_PA8A_URXD | PIO_PA9A_UTXD;
65 
66  //Select CAN0 RX and TX in PIOA
67  uint32_t ul_sr = PIOA->PIO_ABSR;
68  PIOA->PIO_ABSR = ~(PIO_PA1A_CANRX0 | PIO_PA0A_CANTX0) & ul_sr;
69 
70  // Disable the Parallel IO (PIO) of the Rx and Tx pins so that the peripheral controller can use them
71  PIOA->PIO_PDR = PIO_PA1A_CANRX0 | PIO_PA0A_CANTX0;
72 
73  // Enable pull up on CANH and CANL pin
74  PIOA->PIO_PUER = (PIO_PA1A_CANRX0 | PIO_PA0A_CANTX0);
75 
76 
77  //Enable Clock for CAN0 in PMC
78  PMC->PMC_PCR = PMC_PCR_EN | (0 << PMC_PCR_DIV_Pos) | PMC_PCR_CMD | (ID_CAN0 << PMC_PCR_PID_Pos); // DIV = 1(can clk = MCK/2), CMD = 1 (write), PID = 2B (CAN0)
79  PMC->PMC_PCER1 |= 1 << (ID_CAN0 - 32);
80 
81  //Set baudrate, Phase1, phase2 and propagation delay for can bus. Must match on all nodes!
82  CAN0->CAN_BR = can_br;
83 
84 
85  /****** Start of mailbox configuration ******/
86 
87  uint32_t can_ier = 0;
88 
89  /* Configure receive mailboxes */
90  for (int n = num_tx_mb; n <= num_rx_mb + num_tx_mb; n++) //Simply one mailbox setup for all messages. You might want to apply filter for them.
91  {
92  CAN0->CAN_MB[n].CAN_MAM = 0; //Accept all messages
93  CAN0->CAN_MB[n].CAN_MID = CAN_MID_MIDE;
94  CAN0->CAN_MB[n].CAN_MMR = (CAN_MMR_MOT_MB_RX);
95  CAN0->CAN_MB[n].CAN_MCR |= CAN_MCR_MTCR;
96 
97  can_ier |= 1 << n; //Enable interrupt on rx mailbox
98  }
99 
100  /*Configure transmit mailboxes */
101  for (int n = 0; n < num_tx_mb; n++)
102  {
103  CAN0->CAN_MB[n].CAN_MID = CAN_MID_MIDE;
104  CAN0->CAN_MB[n].CAN_MMR = (CAN_MMR_MOT_MB_TX);
105  }
106 
107  /****** End of mailbox configuraion ******/
108 
109  //Enable interrupt on receive mailboxes
110  CAN0->CAN_IER = can_ier;
111 
112  //Enable interrupt in NVIC
113  NVIC_EnableIRQ(ID_CAN0);
114 
115  //enable CAN
116  CAN0->CAN_MR |= CAN_MR_CANEN;
117 
118  return 0;
119 }

130 uint8_t can_send(CAN_MESSAGE* can_msg, uint8_t tx_mb_id)
131 {
132  //Check that mailbox is ready
133  if(CAN0->CAN_MB[tx_mb_id].CAN_MSR & CAN_MSR_MRDY)
134  {
135  //Set message ID and use CAN 2.0B protocol
136  CAN0->CAN_MB[tx_mb_id].CAN_MID = CAN_MID_MIDvA(can_msg->id) | CAN_MID_MIDE ;
137 
138  //Make sure message is not to long
139  if(can_msg->data_length > 7){
140  can_msg->data_length = 7;
141  //Message is to long, sending only the first 8 bytes
142  }
143  //Put message in can data registers
144  CAN0->CAN_MB[tx_mb_id].CAN_MDL = can_msg->data[3] << 24 | can_msg->data[2] << 16 | can_msg->data[1] << 8 | can_msg->data[0];
145  CAN0->CAN_MB[tx_mb_id].CAN_MDH = can_msg->data[7] << 24 | can_msg->data[6] << 16 | can_msg->data[5] << 8 | can_msg->data[4];
146 
147  //Set message length and mailbox ready to send
148  CAN0->CAN_MB[tx_mb_id].CAN_MCR = (can_msg->data_length << CAN_MCR_MDLC_Pos) | CAN_MCR_MTCR;
149  return 0;
150  }
151 
152  else //Mailbox busy
153  {
154  return 1;
155  }
156 
157 }

168 uint8_t can_receive(CAN_MESSAGE* can_msg, uint8_t rx_mb_id)
169 {
170  //Check that mailbox is ready
171  if(CAN0->CAN_MB[rx_mb_id].CAN_MSR & CAN_MSR_MRDY)
172  {
173  //Get data from CAN mailbox
174  uint32_t data_low = CAN0->CAN_MB[rx_mb_id].CAN_MDL;
175  uint32_t data_high = CAN0->CAN_MB[rx_mb_id].CAN_MDH;
176 
177  //Get message ID
178  can_msg->id = (uint16_t)((CAN0->CAN_MB[rx_mb_id].CAN_MID & CAN_MID_MIDvA_Msk) >> CAN_MID_MIDvA_Pos);
179 
180  //Get data length
181  can_msg->data_length = (uint8_t)((CAN0->CAN_MB[rx_mb_id].CAN_MSR & CAN_MSR_MDLC_Msk) >> CAN_MSR_MDLC_Pos);
182 
183  //Put data in CAN_MESSAGE object
184  for(int i = 0; i < can_msg->data_length;i++)
185  {
186  if(i < 4)
187  {
188  can_msg->data[i] = (char)(data_low & 0xff);
189  data_low = data_low >> 8;
190  }
191  else
192  {
193  can_msg->data[i] = (uint8_t)(data_high & 0xff);
194  data_high = data_high >> 8;
195  }
196  }
197 
198  //Reset for new receive
199  CAN0->CAN_MB[rx_mb_id].CAN_MMR = CAN_MMR_MOT_MB_RX;
200  CAN0->CAN_MB[rx_mb_id].CAN_MCR |= CAN_MCR_MTCR;
201  return 0;
202  }
203  else //Mailbox busy
204  {
205  return 1;
206  }
207 }
